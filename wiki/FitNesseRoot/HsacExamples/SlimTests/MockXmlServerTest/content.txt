|add response|{{{<dag/>}}}                 |
|add response|{{{<dag2/>}}}|with status|201|
|$url=       |get mock server url          |

|script|xml http test          |
|post  |{{{<Hallo/>}}} |to|$url|
|check |response status|200    |
